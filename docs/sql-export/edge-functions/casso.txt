import "jsr:@supabase/functions-js/edge-runtime.d.ts";
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2';

console.info('Casso Webhook Server V2 started');

Deno.serve(async (req: Request) => {
  try {
    const supabaseUrl = Deno.env.get('SUPABASE_URL')!;
    const supabaseServiceKey = Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!;
    const cassoToken = Deno.env.get('CASSO_TOKEN')!;

    // Security Check...
    if (req.headers.get('secure-token') !== cassoToken) {
      return new Response(JSON.stringify({ error: 'Unauthorized' }), { status: 401 });
    }

    const body = await req.json();
    if (body.error !== 0) return new Response('No Data', { status: 200 });

    const supabase = createClient(supabaseUrl, supabaseServiceKey);
    const transactions = body.data || [];
    const logs: string[] = [];

    for (const tx of transactions) {
      const description = (tx.description || '').toUpperCase();
      const amount = Number(tx.amount);
      const tid = String(tx.tid);

      // --- CASE 1: MÃ ĐƠN LẺ (CLxxxxxx) ---
      const matchSingle = description.match(/CL[A-Z0-9]{6}/);
      
      // --- CASE 2: MÃ GỘP (GRxxxxxx) ---
      const matchGroup = description.match(/GR[A-Z0-9]{6}/);

      if (matchGroup) {
        // ================= XỬ LÝ GỘP BILL =================
        const groupCode = matchGroup[0];
        console.log(`Found Group Code: ${groupCode}`);

        // 1. Lấy thông tin Group Request
        const { data: groupReq } = await supabase
            .from('group_payment_requests')
            .select('*')
            .eq('group_code', groupCode)
            .single();

        if (groupReq) {
            // 2. Phân bổ tiền (Distribute Logic)
            // Lấy danh sách các khoản nợ trong gói này
            const { data: snapshots } = await supabase
                .from('session_costs_snapshot')
                .select('*')
                .in('id', groupReq.snapshot_ids);

            let remainingMoney = amount;

            // Duyệt qua từng khoản nợ và trả dần
            for (const snap of snapshots || []) {
                if (remainingMoney <= 0) break;

                const debt = Number(snap.final_amount) - Number(snap.paid_amount);
                if (debt <= 0) continue; // Đã trả rồi thì bỏ qua

                // Số tiền trả cho người này = Min(Tiền đang có, Tiền họ nợ)
                const payForThisUser = Math.min(remainingMoney, debt);

                // Insert Payment
                await supabase.from('session_payments').insert({
                    snapshot_id: snap.id,
                    amount: payForThisUser,
                    transaction_id: `${tid}-${snap.id.substring(0,4)}`, // Hack TID unique cho từng sub-payment
                    raw_content: tx.description,
                    note: `Group Payment ${groupCode}`
                });

                // Update Snapshot
                const newPaid = Number(snap.paid_amount) + payForThisUser;
                const newStatus = newPaid >= Number(snap.final_amount) ? 'paid' : 'partial';
                
                await supabase.from('session_costs_snapshot')
                    .update({ paid_amount: newPaid, status: newStatus })
                    .eq('id', snap.id);

                remainingMoney -= payForThisUser;
                logs.push(`Group ${groupCode}: Paid ${payForThisUser} for Snapshot ${snap.id}`);
            }
        } else {
            logs.push(`Ignored: Group Code ${groupCode} expired or not found`);
        }

      } else if (matchSingle) {
        // ================= XỬ LÝ ĐƠN LẺ (Logic Cũ) =================
        const paymentCode = matchSingle[0];
        const { data: snapshot } = await supabase
          .from('session_costs_snapshot')
          .select('*')
          .eq('payment_code', paymentCode)
          .single();

        if (snapshot) {
             const { error: insertErr } = await supabase.from('session_payments').insert({
              snapshot_id: snapshot.id,
              amount: amount,
              transaction_id: tid,
              raw_content: tx.description
            });
            if (!insertErr) {
                // ... Logic update status cũ ...
                const newPaid = Number(snapshot.paid_amount) + amount;
                const newStatus = newPaid >= Number(snapshot.final_amount) ? 'paid' : 'partial';
                await supabase.from('session_costs_snapshot').update({ paid_amount: newPaid, status: newStatus }).eq('id', snapshot.id);
                logs.push(`Single: Updated ${paymentCode}`);
            }
        }
      }
    }

    return new Response(JSON.stringify({ success: true, logs }), { headers: { 'Content-Type': 'application/json' }});
  } catch (err) {
    return new Response(JSON.stringify({ error: err.message }), { status: 500 });
  }
});
